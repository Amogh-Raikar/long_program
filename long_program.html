#include 
using namespace std;

class BINARY {
private:
    string num;

public:
    void read() {
        cout << "Enter a binary number: ";
        cin >> num;
    }

    bool check() {
        for (char c : num) {
            if (c != '0' && c != '1')
                return false;
        }
        return true;
    }

    void onesComplement() {
        for (char &c : num) {
            c = (c == '0') ? '1' : '0';
        }
    }

    void display() {
        cout << "Binary Number: " << num << endl;
    }
};

int main() {
    BINARY b;
    b.read();
    
    if (!b.check()) {
        cout << "Invalid Binary Number!";
        return 0;
    }

    cout << "Original ";
    b.display();

    b.onesComplement();

    cout << "1's Complement ";
    b.display();

    return 0;
}
